LemonCase版本2.0

LemonCase 是一款用于编写WEB工程前端测试用例的编程语言。 简单的仿真指令（动作、等待与断言）语法能更加流畅的描述测试用例。 项目变量与字典变量特性显著提升测试的可维护性。

#TIMES 5
#INTERVAL 3000

process main {
	//需要先登陆后
	//测试创建n个测试项目
	jumpto '/lemonser';
	wait 2000;
	//跳转到我的项目界面
	click "div.selfCard > div.card";
	wait 2000;
	//记录一下当前有几个项目卡
	var projectCard=<#"div.card-container > div.card"/>;
	console projectCard;
	click "span.ico-addRound";
	wait 1000;
	input "div.grid > input.ng-pristine" by |[a-zA-Z]{8}|;
	input "textarea.ng-pristine" by "项目描述"+ |\d{4}|;
	click "div.grid > button:nth-child(5)";
	wait 1000;
	assert <#"div.card-container > div.card"/> > projectCard;
	console "创建项目成功";
}
